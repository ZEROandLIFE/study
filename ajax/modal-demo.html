4. JavaScript 控制弹窗

通过 JavaScript 代码控制弹窗的显示和隐藏

这个弹窗是通过 JavaScript 控制的

// 获取弹窗实例
const jsModal = new bootstrap.Modal(document.getElementById('jsModal'))

// 显示弹窗
function showModal() {
    jsModal.show()
}

// 隐藏弹窗
function hideModal() {
    jsModal.hide()
}

// 事件监听
document.getElementById('jsModal').addEventListener('show.bs.modal', function() {
    console.log('弹窗即将显示')
})
